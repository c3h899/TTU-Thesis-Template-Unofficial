% Verbose Reference Macros
% See https://en.wikibooks.org/wiki/LaTeX/Labels_and_Cross-referencing for additional recommendations
%
% Common Environments
\newcommand{\Algorithm}[1]{Algorithm~\ref{alg:#1}}%
	% Algorithm Reference, Prepends alg:
\newcommand{\Equation}[1]{(\ref{eq:#1})}%
	% IEEE Equation Reference, Prepends eq:
\newcommand{\Figure}[1]{\figurename~\ref{fig:#1}}%
	% Figure Reference, Prepends fig:
\newcommand{\Listing}[1]{Listing~\ref{lst:#1}}%
	% Listing Reference, Prepends lst:
\newcommand{\Table}[1]{Table~\ref{tab:#1}}%
	% Table Reference, Prepends tab:
% Text Sections
\newcommand{\Appendix}[1]{Appendix~\ref{app:#1}}%
	% Appendix Subsection Reference, Prepends app:
\newcommand{\Chapter}[1]{Chapter~\ref{ch:#1}}%
	% Chapter Reference, Prepends ch:
\newcommand{\Paragraph}[1]{Paragraph~\ref{par:#1}}%
	% Paragraph Reference, Prepends par:
\newcommand{\Section}[1]{Section~\ref{sec:#1}}%
	% Section Reference, Prepends sec:
\newcommand{\Subsection}[1]{Subsection~\ref{subsec:#1}}%
	% Subsection Reference, Prepends subsec:
\newcommand{\Subsubsection}[1]{Subsubsection~\ref{subsubsec:#1}}%
	% Subsubsection Reference, Prepends subsubsec:
\newcommand{\Subparagraph}[1]{Subparagraph~\ref{subpar:#1}}%
	% Subparagraph Reference, Prepends subpar:
